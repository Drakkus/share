moro_cav,50,64,5	script	Yellow Seed#bios	CLEAR_NPC,{
	if(!questprogress(15006)) {	
		mes "^ff0000Please talk to the investigator.^000000";
		close;
	}
	.@party_id = getcharid(1);
	.@md_name$ = "Bios Island";
	.@instime = questprogress(15005,PLAYTIME);
	.@rinstime = questprogress(15007,PLAYTIME);
	mes "-Yellow Seed-";
	if(.@instime == 2) {
		mes "-You can look in again-";
		erasequest 15005;
		if(questprogress(15007)) erasequest 15007;
		close;
	}
	else if(.@instime == 1 && .@rinstime != 1) {
		mes "-The passage is closed-";
		mes "-The CD is 23 hours-";
		close;
	}
	if (!.@party_id) {
		mes "You need to create a party.";
		close;
	}
	if(getpartyleader(getcharid(1)) == strcharinfo(0))
		.@str1$ = "Open the Yellow Seed";
	else
		.@str1$ = "";
	.@str2$ = "Enter the Yellow Seed";
	next;
	switch(select(.@str1$,.@str2$,"Leave")) {
	 case 1:
		if(getpartyleader(.@party_id) != strcharinfo(0)) {
			mes "Only the party leader can open the instabce.";
			close;
		}
		if(.@instime){
			mes "-Yellow Seed-";
			mes "-Unable to open instance-";
			mes "-The CD is 23 hours-";
			close;
		}
		.@instance = instance_create(.@md_name$, .@party_id);
		if (.@instance < 0) {
			mes "Party Name:" + getpartyname(.@party_id);
			mes "Part Leader: " + getpartyleader(.@party_id);
			mes "^0000ffBios Island^000000¸Instance failed";
			close;
		}
		if (instance_attachmap("1@dth1",.@instance) != ""&&instance_attachmap("1@dth2",.@instance) != ""&&instance_attachmap("1@dth3",.@instance) != "") {
			instance_set_timeout 5400,300,.@instance;
			instance_init(.@instance);
			mes "-Yellow Seed-";
			mes "The Instance has been opened";
			mes "Please enter right away";
			close;
		} else {
			instance_destroy(.@instance);
			mes "Instance creation failed";
			close;
		}
	 case 2:
		if (has_instance("1@dth1") == "" || has_instance("1@dth2") == "" || has_instance("1@dth3") == "") {
			mes "Can't find an instance for your party, please let the leader open it first.";
			close;
		} 
		mapannounce "moro_cav",getpartyname(getcharid(1))+" Party Member: "+strcharinfo(0)+" Has entered the Bios Island Instance",bc_map,"0x00ffff";
		if(!questprogress(15005))setquest 15005; //CoolDown
		if(!questprogress(15007))setquest 15007; //Re-enter
		close2;
		warp "1@dth1",17,93;
		end;
	 case 3:
		mes "Left the yellow seed";
		close;
	}

OnTimer2000:
	specialeffect EF_ENHANCE;
	end;
OnTimer3600:
	specialeffect EF_ENHANCE;
	end;
OnTimer15000:
	initnpctimer;
	end;

OnInit:
	waitingroom "Bios Island Entrance", 0;
	initnpctimer;
	end; 
}
